-- a) Average salary of instructors in the Computer Science department
SELECT AVG(I.salary) AS AverageSalary
FROM INSTRUCTOR I
JOIN DEPARTMENT D ON I.dept_id = D.dept_id
WHERE D.name = 'Computer Science';

-- b) Average salary in each department
SELECT D.name AS Department, AVG(I.salary) AS AverageSalary
FROM DEPARTMENT D
JOIN INSTRUCTOR I ON D.dept_id = I.dept_id
GROUP BY D.name;

-- c) Total salary of all instructors earning more than 30K
SELECT SUM(salary) AS TotalSalary
FROM INSTRUCTOR
WHERE salary > 30000;

-- d) Number of students enrolled in each course
SELECT C.course_id, C.title, COUNT(T.student_id) AS EnrolledStudents
FROM COURSE C
LEFT JOIN TAKES T ON C.course_id = T.section_id  -- corrected: TAKES references section_id
GROUP BY C.course_id, C.title;

-- e) Departments whose building name includes "Symbio"
SELECT name
FROM DEPARTMENT
WHERE building LIKE '%Symbio%';

-- f) Departments whose name contains "sci" (case-insensitive)
SELECT name
FROM DEPARTMENT
WHERE LOWER(name) LIKE '%sci%';


-- g) Display the length of the name of all the students
SELECT name, LENGTH(name) AS NameLength
FROM STUDENT;

-- h) Retrieve total number of instructors
SELECT COUNT(*) AS TotalInstructors
FROM INSTRUCTOR;

-- Display names of instructors working for the past 5 years
SELECT name
FROM INSTRUCTOR
WHERE hiredate <= ADD_MONTHS(SYSDATE, -60);  -- 60 months = 5 years

-- j) Display instructors who joined before 30th June 1990 or after 31st Dec 1990
SELECT name
FROM INSTRUCTOR
WHERE hiredate < DATE '1990-06-30' OR hiredate > DATE '1990-12-31';

-- k) Display instructors who are going to retire on 01-Jan-2024 (max job period 18 years)
SELECT name
FROM INSTRUCTOR
WHERE hiredate <= DATE '2006-01-01';  -- 18 years before 2024-01-01


-- l) Display instructors whose salary contains at least 4 digits
SELECT name
FROM INSTRUCTOR
WHERE LENGTH(TO_CHAR(FLOOR(salary))) >= 4;

-- m) Display list of instructors with salary categories
SELECT name,
       CASE
           WHEN salary < 60000 THEN 'Less Salary'
           WHEN salary = 60000 THEN 'Exact Salary'
           WHEN salary > 60000 THEN 'More Salary'
       END AS SalaryCategory
FROM INSTRUCTOR;
